Spotify

Download for Windows Download for Linux 
curl -sS https://download.spotify.com/debian/pubkey.gpg | sudo apt-key add -  # Debian/Ubuntu
echo "deb http://repository.spotify.com stable non-free" | sudo tee /etc/apt/sources.list.d/spotify.list
sudo apt update && sudo apt install spotify-client
sudo yum-config-manager --add-repo=http://negativo17.org/repos/epel-spotify.repo  # Red Hat/CentOS
sudo yum install spotify-client
sudo dnf config-manager --add-repo=http://negativo17.org/repos/epel-spotify.repo  # Fedora
sudo dnf install spotify-client
yay -S spotify  # Arch (using AUR helper)
